# Drizzle ORM — SQLite for SuperBad HQ

SuperBad HQ migrates from JSON flat files to SQLite via Drizzle ORM in Session 0.3. All subsequent sessions build on this layer. Never import `better-sqlite3` or query the DB directly — always go through Drizzle.

## Stack

```
better-sqlite3        — SQLite driver (synchronous, perfect for single-user)
drizzle-orm           — Query builder + ORM
drizzle-kit           — Migration CLI
```

Install:
```bash
npm install drizzle-orm better-sqlite3
npm install -D drizzle-kit @types/better-sqlite3
```

## 2. Singleton Connection

```typescript
// src/db/index.ts
import Database from 'better-sqlite3'
import { drizzle } from 'drizzle-orm/better-sqlite3'
import * as schema from './schema'

// Singleton pattern — prevents connection leaks in Next.js dev (hot reload)
const globalForDb = globalThis as unknown as { db: ReturnType<typeof drizzle> }

const sqlite = new Database('data/superbad.db')

// Enable WAL mode — significantly faster for concurrent reads
sqlite.pragma('journal_mode = WAL')
sqlite.pragma('foreign_keys = ON')

export const db = globalForDb.db ?? drizzle(sqlite, { schema })

if (process.env.NODE_ENV !== 'production') globalForDb.db = db
```

## 3. Schema Definition

```typescript
// src/db/schema/clients.ts
import { sqliteTable, text, integer, real } from 'drizzle-orm/sqlite-core'
import { relations } from 'drizzle-orm'

export const clients = sqliteTable('clients', {
  id: text('id').primaryKey().$defaultFn(() => crypto.randomUUID()),
  name: text('name').notNull(),
  company: text('company'),
  email: text('email').notNull().unique(),
  phone: text('phone'),                          // indexed — used for Twilio inbound matching
  status: text('status', {
    enum: ['trial', 'retainer', 'flagship', 'churned', 'prospect']
  }).notNull().default('prospect'),
  tier: text('tier', {
    enum: ['trial_shoot', 'performance_retainer', 'flagship']
  }),
  monthlyValue: real('monthly_value'),
  vertical: text('vertical', {
    enum: ['medical_aesthetics', 'financial', 'allied_health', 'other']
  }),
  notes: text('notes'),
  createdAt: integer('created_at', { mode: 'timestamp' }).$defaultFn(() => new Date()),
  updatedAt: integer('updated_at', { mode: 'timestamp' }).$defaultFn(() => new Date()),
})

// Relations
export const clientsRelations = relations(clients, ({ many }) => ({
  contacts: many(contacts),
  deals: many(deals),
  communications: many(communications),
  activities: many(activityLog),
}))

// TypeScript types — ALWAYS infer, never write manually
export type Client = typeof clients.$inferSelect
export type NewClient = typeof clients.$inferInsert
```

## 5. Migration Commands

```bash
# Generate migration files from schema changes
npx drizzle-kit generate

# Apply pending migrations to the database
npx drizzle-kit migrate

# Open the Drizzle Studio GUI (useful for inspecting data)
npx drizzle-kit studio

# Push schema directly to DB (dev only — skips migration files)
npx drizzle-kit push
```

**Rule:** Always use `generate` + `migrate` (never `push`) in production. Commit migration files to git.

Add to `package.json`:
```json
"scripts": {
  "db:generate": "drizzle-kit generate",
  "db:migrate": "drizzle-kit migrate",
  "db:studio": "drizzle-kit studio",
  "db:push": "drizzle-kit push"
}
```

## 6. Query Patterns

```typescript
import { db } from '@/db'
import { clients, deals } from '@/db/schema'
import { eq, and, like, desc, asc, count, sql } from 'drizzle-orm'

// --- SELECT ---

// Get all clients
const allClients = await db.select().from(clients)

// Get one by ID
const client = await db.select().from(clients).where(eq(clients.id, id)).get()
// Note: .get() returns one row or undefined; .all() returns array

// Filter + sort
const retainerClients = await db
  .select()
  .from(clients)
  .where(eq(clients.status, 'retainer'))
  .orderBy(desc(clients.createdAt))

// Search across fields
const results = await db
  .select()
  .from(clients)
  .where(like(clients.name, `%${query}%`))

// Join with relations (using query API — cleaner than raw joins)
const clientWithDeals = await db.query.clients.findFirst({
  where: eq(clients.id, id),
  with: {
    deals: true,
    communications: { orderBy: desc(communications.createdAt), limit: 10 },
  },
})

// --- INSERT ---

const newClient = await db.insert(clients).values({
  name: 'Dr Sarah Chen',
  email: 'sarah@skinclinic.com.au',
  status: 'prospect',
  vertical: 'medical_aesthetics',
}).returning()

// --- UPDATE ---

await db
  .update(clients)
  .set({ status: 'retainer', updatedAt: new Date() })
  .where(eq(clients.id, id))

// --- DELETE ---

await db.delete(clients).where(eq(clients.id, id))

// --- AGGREGATE ---

const [{ total }] = await db
  .select({ total: count() })
  .from(clients)
  .where(eq(clients.status, 'retainer'))
```

## 7. In API Routes (Next.js App Router)

```typescript
// src/app/api/clients/route.ts
import { NextResponse } from 'next/server'
import { db } from '@/db'
import { clients } from '@/db/schema'
import { clientSchema } from '@/lib/schemas/client'

export async function GET() {
  const allClients = await db.select().from(clients).orderBy(desc(clients.createdAt))
  return NextResponse.json(allClients)
}

export async function POST(request: Request) {
  const body = await request.json()
  const parsed = clientSchema.safeParse(body)

  if (!parsed.success) {
    return NextResponse.json(
      { errors: parsed.error.flatten().fieldErrors },
      { status: 422 }
    )
  }

  const [client] = await db.insert(clients).values(parsed.data).returning()
  return NextResponse.json(client, { status: 201 })
}
```

## 8. JSON → SQLite Migration Script Pattern

```typescript
// scripts/migrate-json.ts  (run once: npx tsx scripts/migrate-json.ts)
import { db } from '../src/db'
import { clients } from '../src/db/schema'
import clientsJson from '../data/clients.json'

async function migrateClients() {
  console.log(`Migrating ${clientsJson.length} clients...`)

  for (const legacy of clientsJson) {
    await db.insert(clients).values({
      id: legacy.id ?? crypto.randomUUID(),
      name: legacy.name,
      email: legacy.email ?? `unknown-${Date.now()}@placeholder.com`,
      status: mapLegacyStatus(legacy.status),
      // ...map other fields
      createdAt: legacy.createdAt ? new Date(legacy.createdAt) : new Date(),
    }).onConflictDoNothing()  // safe to re-run
  }

  console.log('Done.')
}

function mapLegacyStatus(status: string): 'trial' | 'retainer' | 'flagship' | 'churned' | 'prospect' {
  const map: Record<string, 'trial' | 'retainer' | 'flagship' | 'churned' | 'prospect'> = {
    'active': 'retainer',
    'lead': 'prospect',
    'inactive': 'churned',
  }
  return map[status] ?? 'prospect'
}

migrateClients()
```

## 10. Critical Rules

- **Never use `.push` in production** — always generate + migrate
- **Always enable WAL mode** — set `journal_mode = WAL` on connection
- **Always enable foreign keys** — set `foreign_keys = ON` on connection
- **Phone field must be indexed** for Twilio inbound SMS matching: add `.index('idx_clients_phone', (t) => t.phone)` to the table definition
- **Never write TypeScript types manually** — always use `$inferSelect` / `$inferInsert`
- **Migrations are committed to git** — never `.gitignore` the migrations folder
- **Use `.get()` not `.all()[0]`** when fetching a single row — returns `undefined` cleanly on no match
